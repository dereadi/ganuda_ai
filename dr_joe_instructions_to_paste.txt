from telegram.ext import Application, MessageHandler, filters
import requests

BOT_TOKEN = "YOUR_TOKEN_HERE"

async def handle_message(update, context):
    text = update.message.text
    if 'bigmac' in text.lower() or 'council' in text.lower():
        # Query your Ollama
        resp = requests.post(
            "http://localhost:8000/api/generate",
            json={"model": "llama3.1", "prompt": text, "stream": False}
        )
        result = resp.json().get('response', 'No response')
        await update.message.reply_text(f"🏔️ BigMac Council: {result}")

app = Application.builder().token(BOT_TOKEN).build()
app.add_handler(MessageHandler(filters.TEXT, handle_message))
print("✅ Bot ready! Add it to Ganuda-BotComms channel as admin")
app.run_polling()
