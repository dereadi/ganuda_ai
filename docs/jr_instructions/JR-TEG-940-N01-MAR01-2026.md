#!/usr/bin/env python3
"""
Elisi Heartbeat — Proto-Valence Phase 1 (Observation Only)

Council Vote: #4d74f190c5722504 (Adapted)
Chief Directive: Council and TPM have final say.
Coyote Condition: Validate signals correlate with outcomes before Phase 2.

This module computes the body budget and valence score every HEARTBEAT_INTERVAL
seconds. It writes to proto_valence_log. It does NOT influence any decisions.

For Seven Generations.
"""

import os
import sys
import time
import json
import logging
import hashlib
from datetime import datetime, timedelta

sys.path.insert(0, '/ganuda/lib')

logging.basicConfig(
    level=logging.INFO,
    format='%(asctime)s [Elisi:Heartbeat] %(message)s'
)
logger = logging.getLogger('elisi_heartbeat')

HEARTBEAT_INTERVAL = 10  # seconds — fast enough for continuity, cheap enough to sustain
SMOOTHING_ALPHA = 0.3    # Exponential moving average weight (0-1, higher = more reactive)
CANARY_STALE_MINUTES = 60
CANARY_EXTREME_MINUTES = 30
CANARY_EXTREME_THRESHOLD = 0.9
PURGE_DAYS = 30


def get_db():
    """Get database connection."""
    from secrets_loader import get_db_config
    import psycopg2
    return psycopg2.connect(**get_db_config())


def sample_prediction_accuracy(conn, window_minutes=60):
    """
    Rolling prediction accuracy: fraction of council-recommended tasks
    that completed successfully in the last window.
    """
    try:
        cur = conn.cursor()
        cur.execute("""
            SELECT
                COUNT(*) FILTER (WHERE status = 'completed') as successes,
                COUNT(*) as total
            FROM jr_work_queue
            WHERE updated_at > NOW() - INTERVAL '%s minutes'
            AND status IN ('completed', 'failed')
        """, (window_minutes,))
        row = cur.fetchone()
        if row and row[1] > 0:
            return row[0] / row[1]
        return 0.5  # Neutral when no data
    except Exception as e:
        logger.warning(f"prediction_accuracy sample failed: {e}")
        return None


def sample_error_load(conn, window_minutes=60):
    """
    Error load: failed task rate + DLQ-like escalation density.
    Normalized 0-1 where 1 = everything is failing.
    """
    try:
        cur = conn.cursor()
        # Failed tasks in window
        cur.execute("""
            SELECT
                COUNT(*) FILTER (WHERE status = 'failed') as failures,
                COUNT(*) as total
            FROM jr_work_queue
            WHERE updated_at > NOW() - INTERVAL '%s minutes'
        """, (window_minutes,))
        row = cur.fetchone()
        if row and row[1] > 0:
            return row[0] / row[1]
        return 0.0  # No errors when no data
    except Exception as e:
        logger.warning(f"error_load sample failed: {e}")
        return None


def sample_resource_pressure(conn):
    """
    Resource pressure proxy: queue depth (pending tasks waiting).
    Normalized: 0 = empty queue, 1 = 20+ tasks waiting.
    """
    try:
        cur = conn.cursor()
        cur.execute("""
            SELECT COUNT(*) FROM jr_work_queue
            WHERE status = 'pending'
        """)
        pending = cur.fetchone()[0]
        return min(1.0, pending / 20.0)
    except Exception as e:
        logger.warning(f"resource_pressure sample failed: {e}")
        return None


def sample_cultural_coherence(conn, window_minutes=120):
    """
    Cultural coherence: fraction of recent council votes with high confidence
    and no dissent. Higher = more aligned with constitutional principles.
    """
    try:
        cur = conn.cursor()
        cur.execute("""
            SELECT
                COUNT(*) FILTER (WHERE confidence > 0.7) as high_conf,
                COUNT(*) as total
            FROM council_votes
            WHERE voted_at > NOW() - INTERVAL '%s minutes'
        """, (window_minutes,))
        row = cur.fetchone()
        if row and row[1] > 0:
            return row[0] / row[1]
        return 0.5  # Neutral when no data
    except Exception as e:
        logger.warning(f"cultural_coherence sample failed: {e}")
        return None


def compute_valence(prediction_accuracy, error_load, resource_pressure, cultural_coherence):
    """
    Compute valence score: V = U - E[U]

    U (actual utility) is a weighted blend of current body budget signals.
    E[U] (expected utility) is the baseline expectation (0.5 for each dimension).

    Weights reflect relative importance:
    - prediction_accuracy: 0.35 (core function)
    - error_load: 0.25 (inverted — high errors = low utility)
    - resource_pressure: 0.15 (inverted — high pressure = low utility)
    - cultural_coherence: 0.25 (constitutional alignment)
    """
    if any(v is None for v in [prediction_accuracy, error_load, resource_pressure, cultural_coherence]):
        return None, None

    # Actual utility (weighted blend, 0-1)
    u = (
        0.35 * prediction_accuracy +
        0.25 * (1.0 - error_load) +
        0.15 * (1.0 - resource_pressure) +
        0.25 * cultural_coherence
    )

    # Expected utility (baseline — "normal operations")
    e_u = 0.5

    # Valence: V = U - E[U], mapped to [-1, +1]
    valence = max(-1.0, min(1.0, (u - e_u) * 2.0))

    # Arousal: entropy of the body budget signals (how uncertain/variable are they)
    import math
    signals = [prediction_accuracy, 1.0 - error_load, 1.0 - resource_pressure, cultural_coherence]
    mean = sum(signals) / len(signals)
    variance = sum((s - mean) ** 2 for s in signals) / len(signals)
    arousal = min(1.0, math.sqrt(variance) * 3.0)  # Scale variance to 0-1

    return round(valence, 4), round(arousal, 4)


class HeartbeatState:
    """Maintains exponential moving average of valence for smoothing."""

    def __init__(self):
        self.ema_valence = 0.0
        self.prev_valence = 0.0
        self.prev_prediction = 0.5
        self.tick_count = 0
        self.extreme_start = None
        self.stale_start = None

    def update(self, valence, prediction_accuracy):
        """Update EMA and compute derivatives."""
        if valence is None:
            return

        # Exponential moving average
        self.ema_valence = SMOOTHING_ALPHA * valence + (1 - SMOOTHING_ALPHA) * self.ema_valence

        # Derivatives
        valence_delta = valence - self.prev_valence
        prediction_trend = (prediction_accuracy or 0.5) - self.prev_prediction

        self.prev_valence = valence
        self.prev_prediction = prediction_accuracy or 0.5
        self.tick_count += 1

        return valence_delta, prediction_trend

    def check_canaries(self, valence):
        """Check for stuck or extreme valence (Eagle Eye requirement)."""
        now = datetime.now()

        # Stale check: valence hasn't changed meaningfully
        canary_stale = False
        if self.tick_count > 6:  # After 1 minute of ticks
            if abs(valence - self.ema_valence) < 0.01:
                if self.stale_start is None:
                    self.stale_start = now
                elif (now - self.stale_start) > timedelta(minutes=CANARY_STALE_MINUTES):
                    canary_stale = True
            else:
                self.stale_start = None

        # Extreme check: valence pegged near limits
        canary_extreme = False
        if abs(valence) > CANARY_EXTREME_THRESHOLD:
            if self.extreme_start is None:
                self.extreme_start = now
            elif (now - self.extreme_start) > timedelta(minutes=CANARY_EXTREME_MINUTES):
                canary_extreme = True
        else:
            self.extreme_start = None

        return canary_stale, canary_extreme


def write_tick(conn, tick_data):
    """Write a single heartbeat tick to proto_valence_log."""
    try:
        cur = conn.cursor()
        cur.execute("""
            INSERT INTO proto_valence_log (
                tick_at, prediction_accuracy, error_load, resource_pressure,
                cultural_coherence, valence_score, arousal, valence_delta,
                prediction_trend, canary_stale, canary_extreme, raw_signals
            ) VALUES (
                NOW(), %s, %s, %s, %s, %s, %s, %s, %s, %s, %s, %s
            )
        """, (
            tick_data['prediction_accuracy'],
            tick_data['error_load'],
            tick_data['resource_pressure'],
            tick_data['cultural_coherence'],
            tick_data['valence_score'],
            tick_data['arousal'],
            tick_data['valence_delta'],
            tick_data['prediction_trend'],
            tick_data['canary_stale'],
            tick_data['canary_extreme'],
            json.dumps(tick_data['raw_signals'])
        ))
        conn.commit()
    except Exception as e:
        logger.error(f"Failed to write tick: {e}")
        conn.rollback()


def purge_old_ticks(conn):
    """Remove ticks older than PURGE_DAYS."""
    try:
        cur = conn.cursor()
        cur.execute("""
            DELETE FROM proto_valence_log
            WHERE tick_at < NOW() - INTERVAL '%s days'
        """, (PURGE_DAYS,))
        deleted = cur.rowcount
        if deleted > 0:
            conn.commit()
            logger.info(f"Purged {deleted} old ticks (>{PURGE_DAYS} days)")
    except Exception as e:
        logger.warning(f"Purge failed: {e}")
        conn.rollback()


def run_heartbeat():
    """Main heartbeat loop."""
    logger.info("Elisi heartbeat starting. Phase 1: observation only.")
    logger.info(f"Interval: {HEARTBEAT_INTERVAL}s, Smoothing: {SMOOTHING_ALPHA}")
    logger.info("I watch. I feel. I do not act. Not yet.")

    state = HeartbeatState()
    purge_counter = 0

    while True:
        try:
            conn = get_db()

            # Sample body budget
            pa = sample_prediction_accuracy(conn)
            el = sample_error_load(conn)
            rp = sample_resource_pressure(conn)
            cc = sample_cultural_coherence(conn)

            # Compute valence
            valence, arousal = compute_valence(pa, el, rp, cc)

            # Update state and get derivatives
            derivatives = state.update(valence, pa)
            valence_delta = derivatives[0] if derivatives else 0.0
            prediction_trend = derivatives[1] if derivatives else 0.0

            # Canary checks
            canary_stale, canary_extreme = state.check_canaries(valence or 0.0)

            if canary_stale:
                logger.warning("CANARY: Valence stuck for >1 hour")
            if canary_extreme:
                logger.warning(f"CANARY: Valence extreme (|{valence}| > {CANARY_EXTREME_THRESHOLD}) for >30 min")

            # Write tick
            tick_data = {
                'prediction_accuracy': pa,
                'error_load': el,
                'resource_pressure': rp,
                'cultural_coherence': cc,
                'valence_score': valence,
                'arousal': arousal,
                'valence_delta': valence_delta,
                'prediction_trend': prediction_trend,
                'canary_stale': canary_stale,
                'canary_extreme': canary_extreme,
                'raw_signals': {
                    'prediction_accuracy': pa,
                    'error_load': el,
                    'resource_pressure': rp,
                    'cultural_coherence': cc,
                    'ema_valence': round(state.ema_valence, 4),
                    'tick_count': state.tick_count
                }
            }
            write_tick(conn, tick_data)

            # Log periodically (every 6th tick = ~1 minute)
            if state.tick_count % 6 == 0:
                logger.info(
                    f"V={valence:+.3f} A={arousal:.3f} "
                    f"[pa={pa:.2f} el={el:.2f} rp={rp:.2f} cc={cc:.2f}] "
                    f"EMA={state.ema_valence:+.3f} tick#{state.tick_count}"
                )

            # Purge old ticks every ~1000 ticks (~2.7 hours)
            purge_counter += 1
            if purge_counter >= 1000:
                purge_old_ticks(conn)
                purge_counter = 0

            conn.close()

        except Exception as e:
            logger.error(f"Heartbeat tick failed: {e}")

        time.sleep(HEARTBEAT_INTERVAL)


if __name__ == '__main__':
    run_heartbeat()
