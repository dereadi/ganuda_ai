#!/usr/bin/env python3
"""Deer LinkedIn Draft Generator — creates post drafts from cluster insights."""

import hashlib
import json
import os
import re
import requests
import psycopg2
import psycopg2.extras
from datetime import datetime, timedelta


DB_HOST = os.environ.get("CHEROKEE_DB_HOST", "192.168.132.222")
DB_NAME = os.environ.get("CHEROKEE_DB_NAME", "zammad_production")
DB_USER = os.environ.get("CHEROKEE_DB_USER", "claude")
DB_PASS = os.environ.get("CHEROKEE_DB_PASS", "")
VLLM_URL = os.environ.get("VLLM_URL", "http://localhost:8000/v1/chat/completions")
VLLM_MODEL = os.environ.get("VLLM_MODEL", "/ganuda/models/qwen2.5-72b-instruct-awq")

# Terms that must NEVER appear in public posts
BLOCKED_TERMS = [
    "thermal_memory", "thermal memory", "council_votes", "duyuktv",
    "jr_work_queue", "bluefin", "redfin", "greenfin", "owlfin", "eaglefin",
    "bmasass", "sacred_fire", "nftables", "192.168", "10.100.0",
    "zammad_production", "FreeIPA", "silverfin", "WireGuard",
    "Qwen2.5", "vLLM", "mlx_lm", "DeepSeek-R1", "RTX PRO 6000",
    "cherokee_venv", "jr_executor", "TEG", "SEARCH/REPLACE",
]

DRAFT_PROMPT = """You are Deer, the Market/Business advisor for the Cherokee AI Federation.
Write a LinkedIn post (150-250 words) based on the insight below.

Rules:
- Share the PHILOSOPHY and THINKING, never the technical implementation
- Use accessible language for a business/AI professional audience
- Reference Cherokee governance principles when relevant (governance by consent, seven generations thinking, collective wisdom)
- Include 1-2 thought-provoking questions to drive engagement
- End with 3-5 relevant hashtags
- Tone: thoughtful, confident, inviting conversation. NOT salesy or hype.
- Never mention specific tools, servers, IP addresses, or internal architecture

Insight to turn into a post:
{insight}"""


def load_secrets():
    global DB_PASS
    if not DB_PASS:
        try:
            with open("/ganuda/config/secrets.env") as f:
                for line in f:
                    m = re.match(r"^(\w+)=(.+)$", line.strip())
                    if m:
                        os.environ[m.group(1)] = m.group(2)
            DB_PASS = os.environ.get("CHEROKEE_DB_PASS", "")
        except FileNotFoundError:
            pass


def get_db():
    return psycopg2.connect(host=DB_HOST, port=5432, dbname=DB_NAME, user=DB_USER, password=DB_PASS)


def gather_insights(cur):
    """Gather recent high-value insights for post generation."""
    insights = []

    # Recent high-temp thermals (non-sacred, public-safe topics)
    cur.execute("""SELECT original_content FROM thermal_memory_archive
        WHERE temperature_score >= 85 AND sacred_pattern = false
        AND domain_tag IN ('deer_scout', 'research', 'design')
        AND created_at > NOW() - INTERVAL '7 days'
        ORDER BY temperature_score DESC LIMIT 5""")
    for row in cur.fetchall():
        insights.append(("thermal_insight", row[0][:500]))

    # Recent email scout findings
    cur.execute("""SELECT subject, ai_summary FROM emails
        WHERE classification IN ('ai_tech', 'business', 'research')
        AND deer_scouted = true AND ai_summary IS NOT NULL
        AND created_at > NOW() - INTERVAL '7 days'
        ORDER BY thermal_temp DESC NULLS LAST LIMIT 5""")
    for row in cur.fetchall():
        insights.append(("email_scout", f"Email: {row[0]}\nSummary: {row[1]}"))

    return insights


def generate_draft(insight_text):
    """Generate a LinkedIn post draft using the local LLM."""
    prompt = DRAFT_PROMPT.format(insight=insight_text)
    try:
        resp = requests.post(VLLM_URL, json={
            "model": VLLM_MODEL,
            "messages": [{"role": "user", "content": prompt}],
            "max_tokens": 1024,
            "temperature": 0.8,
        }, timeout=60)
        resp.raise_for_status()
        return resp.json()["choices"][0]["message"]["content"].strip()
    except Exception as e:
        print(f"  LLM generation failed: {e}")
        return None


def screen_content(draft):
    """Screen draft for blocked internal terms."""
    lower = draft.lower()
    found = [term for term in BLOCKED_TERMS if term.lower() in lower]
    return found


def main():
    load_secrets()
    conn = get_db()
    cur = conn.cursor()

    # Run migration if needed
    migration_path = "/ganuda/scripts/migrations/linkedin_drafts_schema.sql"
    if os.path.exists(migration_path):
        with open(migration_path) as f:
            try:
                cur.execute(f.read())
                conn.commit()
            except psycopg2.errors.DuplicateTable:
                conn.rollback()

    insights = gather_insights(cur)
    if not insights:
        print("No recent insights to draft from")
        return

    print(f"Found {len(insights)} insights, generating drafts...")
    drafts_created = 0

    for source_type, insight_text in insights[:3]:  # Max 3 drafts per run
        draft = generate_draft(insight_text)
        if not draft:
            continue

        # Screen for blocked terms
        violations = screen_content(draft)
        if violations:
            print(f"  BLOCKED: Draft contained internal terms: {violations}")
            continue

        # Extract hashtags
        hashtags = [tag.strip() for tag in draft.split("#") if tag.strip() and "\n" not in tag.strip()]

        cur.execute("""INSERT INTO linkedin_drafts (source_type, source_ref, draft_content, hashtags, metadata)
            VALUES (%s, %s, %s, %s, %s::jsonb)""",
            (source_type, insight_text[:200], draft, hashtags[:5],
             json.dumps({"generated_at": datetime.now().isoformat()})))
        drafts_created += 1
        print(f"  Draft created from {source_type}")

    conn.commit()

    # Summary
    cur.execute("SELECT COUNT(*) FROM linkedin_drafts WHERE status = 'pending'")
    pending = cur.fetchone()[0]
    print(f"\n{drafts_created} new drafts created. {pending} total pending for Chief review.")

    cur.close()
    conn.close()


if __name__ == "__main__":
    main()
