## Steps

### Step 1: Create proto_valence_log table

File: `scripts/migrations/proto_valence_schema.sql`

Create `scripts/migrations/proto_valence_schema.sql`

```sql
-- Proto-Valence Phase 1: Observation-Only Body Budget
-- Council Vote #4d74f190c5722504 (Adapted)
-- READ-ONLY dependencies: council_votes, jr_work_queue, thermal_memory_archive
-- WRITES ONLY to: proto_valence_log (this table)
-- Does NOT modify sacred fire, thermal memory, or council deliberation

CREATE TABLE IF NOT EXISTS proto_valence_log (
    id SERIAL PRIMARY KEY,

    -- Heartbeat timestamp
    tick_at TIMESTAMPTZ NOT NULL DEFAULT NOW(),

    -- Body Budget dimensions (raw signals)
    prediction_accuracy FLOAT,      -- Rolling council recommendation outcome rate
    error_load FLOAT,               -- DLQ depth + failed task rate (0-1 normalized)
    resource_pressure FLOAT,        -- GPU util + queue depth proxy (0-1 normalized)
    cultural_coherence FLOAT,       -- Sacred pattern density in recent decisions (0-1)

    -- Computed valence (V = U - E[U])
    valence_score FLOAT,            -- Scalar approach/avoid (-1.0 to +1.0)
    arousal FLOAT,                  -- Entropy of posterior beliefs (0-1)

    -- Derivatives (compression progress)
    valence_delta FLOAT,            -- Change from previous tick
    prediction_trend FLOAT,         -- First derivative of prediction_accuracy

    -- Canary fields (Eagle Eye requirement)
    canary_stale BOOLEAN DEFAULT FALSE,   -- True if valence stuck >1hr
    canary_extreme BOOLEAN DEFAULT FALSE, -- True if |valence| > 0.9 for >30min

    -- Metadata
    tick_source VARCHAR(50) DEFAULT 'elisi_heartbeat',
    raw_signals JSONB,              -- Full snapshot of all input signals

    -- Indexing
    created_at TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

-- Time-series index for dashboard queries
CREATE INDEX IF NOT EXISTS idx_pvl_tick_at ON proto_valence_log (tick_at DESC);

-- Canary alert index
CREATE INDEX IF NOT EXISTS idx_pvl_canary ON proto_valence_log (canary_stale, canary_extreme) WHERE canary_stale = true OR canary_extreme = true;

-- Keep 30 days of ticks (at 10s intervals = ~260K rows/month)
-- Purge policy: Elisi handles via age-based DELETE in heartbeat loop

COMMENT ON TABLE proto_valence_log IS 'Proto-valence Phase 1: observation-only body budget. Council vote #4d74f190c5722504. READ-ONLY deps on council_votes, jr_work_queue, thermal_memory_archive. Does NOT influence decisions.';
```
