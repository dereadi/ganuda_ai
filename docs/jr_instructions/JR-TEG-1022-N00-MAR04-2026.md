The Graduated Harness is a three-tier inference architecture:
- **Tier 1 (Reflex)**: ~50ms, single LLM call, config-driven, no federation deps
- **Tier 2 (Deliberation)**: ~500ms, 2-3 specialists, quick consensus
- **Tier 3 (Council)**: ~30-120s, full ethos, sacred patterns, thermal memory

This instruction builds the **shared foundation**: dataclasses, configuration loader, and the escalation engine that routes requests through tiers based on confidence and stakes detection (DC-6 Gradient Principle). Users never choose tiers; the system finds its own level.

**Design Constraints Applied**:
- DC-5 (Coyote as Cam): Low-cost observation at every tier
- DC-6 (Gradient Principle): Specialization is gravity, not boundary
- DC-7 (Noyawisgi): Transform under pressure, not restore

**Key Decision**: No de-escalation in v1. Once escalated, a request stays at the higher tier.

```python
"""
Harness Core -- Request/Response models and tier result types.

These dataclasses are the shared language across all three tiers.
Every tier returns a HarnessResponse. The escalation engine reads
confidence and stakes to decide whether to promote to the next tier.

Design note: dataclasses over Pydantic for zero external dependencies.
The harness core must be importable on any node without pip install.
"""

from dataclasses import dataclass, field
from enum import Enum
from typing import Optional, List, Dict, Any
import time
import uuid


class StakesLevel(str, Enum):
    """Stakes classification for escalation decisions.

    Coyote rule: humans underestimate stakes.
    When in doubt, escalate. Cost of over-escalation < cost of under-escalation.
    """
    LOW = "low"
    MEDIUM = "medium"
    HIGH = "high"


@dataclass
class HarnessRequest:
    """Unified request object for all harness tiers.

    Attributes:
        query: The user's question or instruction.
        context: Arbitrary context dict (calendar data, email metadata, etc.).
        user_id: For personalization and rate limiting (Deer requirement).
        session_id: For conversation continuity across requests.
        request_id: Unique ID for tracing. Auto-generated if not provided.
        force_tier: Only for testing -- production NEVER sets this.
        metadata: Additional metadata for routing decisions.
    """
    query: str
    context: Dict[str, Any] = field(default_factory=dict)
    user_id: str = "default"
    session_id: str = ""
    request_id: str = ""
    force_tier: Optional[int] = None
    metadata: Dict[str, Any] = field(default_factory=dict)

    def __post_init__(self):
        if not self.request_id:
            self.request_id = uuid.uuid4().hex[:16]
        if not self.session_id:
            self.session_id = uuid.uuid4().hex[:16]

    def validate(self) -> List[str]:
        """Validate request fields. Returns list of error messages (empty = valid).

        Crawdad requirement: input validation at every tier boundary.
        """
        errors = []
        if not self.query or not isinstance(self.query, str):
            errors.append("query must be a non-empty string")
        if len(self.query) > 50000:
            errors.append("query exceeds maximum length of 50000 characters")
        if self.force_tier is not None and self.force_tier not in (1, 2, 3):
            errors.append("force_tier must be 1, 2, or 3")
        if not isinstance(self.context, dict):
            errors.append("context must be a dict")
        # Sanitize: strip null bytes from query (injection prevention)
        if isinstance(self.query, str):
            self.query = self.query.replace('\x00', '')
        return errors


@dataclass
class TierResult:
    """Result from a single tier's processing.

    Used internally by the escalation engine to track per-tier outcomes.
    The final HarnessResponse is built from the last tier's result plus
    the full escalation path.
    """
    tier: int
    answer: str
    confidence: float
    latency_ms: float
    stakes: StakesLevel = StakesLevel.LOW
    diversity_score: Optional[float] = None
    council_vote_id: Optional[str] = None
    specialist_count: int = 0
    metadata: Dict[str, Any] = field(default_factory=dict)


@dataclass
class HarnessResponse:
    """Unified response from the harness.

    Every request gets exactly one HarnessResponse, regardless of how many
    tiers it passed through. The escalation_path shows the journey.

    Attributes:
        answer: The final answer text.
        tier_used: The tier that produced the final answer (1, 2, or 3).
        confidence: Confidence score from the final tier (0.0 - 1.0).
        escalation_path: List of tiers traversed, e.g. [1, 2] means T1 escalated to T2.
        latency_ms: Total wall-clock latency across all tiers.
        diversity_score: Specialist diversity metric (Tier 2+ only, Elisi requirement).
        council_vote_id: Council vote reference (Tier 3 only).
        request_id: Echoed from the request for tracing.
        metadata: Tier-specific metadata and escalation reasoning.
    """
    answer: str
    tier_used: int
    confidence: float
    escalation_path: List[int] = field(default_factory=list)
    latency_ms: float = 0.0
    diversity_score: Optional[float] = None
    council_vote_id: Optional[str] = None
    request_id: str = ""
    metadata: Dict[str, Any] = field(default_factory=dict)

    def to_dict(self) -> Dict[str, Any]:
        """Serialize to dict for JSON responses."""
        return {
            "answer": self.answer,
            "tier_used": self.tier_used,
            "confidence": self.confidence,
            "escalation_path": self.escalation_path,
            "latency_ms": round(self.latency_ms, 2),
            "diversity_score": self.diversity_score,
            "council_vote_id": self.council_vote_id,
            "request_id": self.request_id,
            "metadata": self.metadata,
        }
```

```python
"""
Harness Configuration -- YAML-driven tier configuration.

All tier behavior is config-driven: endpoints, thresholds, timeouts, prompt
templates, and escalation parameters. Changing the model, the confidence
threshold, or the timeout requires only a YAML edit -- no code changes.

Spider requirement: Tier 1 must be self-contained (no federation deps).
Turtle requirement: No vendor lock-in at Tier 1 (swap models via config).
"""

import os
import yaml
import logging
from dataclasses import dataclass, field
from typing import Dict, Optional, List, Any

logger = logging.getLogger("harness.config")


@dataclass
class EndpointConfig:
    """Configuration for an LLM endpoint (OpenAI-compatible API)."""
    url: str = "http://localhost:8000/v1/chat/completions"
    model: str = "default"
    api_key: str = ""
    timeout_seconds: int = 30
    max_tokens: int = 2048
    temperature: float = 0.7

    @classmethod
    def from_dict(cls, d: Dict[str, Any]) -> "EndpointConfig":
        return cls(
            url=d.get("url", "http://localhost:8000/v1/chat/completions"),
            model=d.get("model", "default"),
            api_key=d.get("api_key", ""),
            timeout_seconds=d.get("timeout_seconds", 30),
            max_tokens=d.get("max_tokens", 2048),
            temperature=d.get("temperature", 0.7),
        )


@dataclass
class TierConfig:
    """Configuration for a single harness tier."""
    enabled: bool = True
    primary_endpoint: EndpointConfig = field(default_factory=EndpointConfig)
    fallback_endpoint: Optional[EndpointConfig] = None
    confidence_threshold: float = 0.7
    latency_target_ms: int = 50
    max_specialists: int = 1
    prompt_template: str = "You are a helpful assistant. Answer the following question:\n\n{query}"

    @classmethod
    def from_dict(cls, d: Dict[str, Any]) -> "TierConfig":
        primary = EndpointConfig.from_dict(d.get("primary_endpoint", {}))
        fallback = None
        if "fallback_endpoint" in d and d["fallback_endpoint"]:
            fallback = EndpointConfig.from_dict(d["fallback_endpoint"])
        return cls(
            enabled=d.get("enabled", True),
            primary_endpoint=primary,
            fallback_endpoint=fallback,
            confidence_threshold=d.get("confidence_threshold", 0.7),
            latency_target_ms=d.get("latency_target_ms", 50),
            max_specialists=d.get("max_specialists", 1),
            prompt_template=d.get("prompt_template",
                "You are a helpful assistant. Answer the following question:\n\n{query}"),
        )


@dataclass
class EscalationConfig:
    """Configuration for the escalation engine."""
    tier1_confidence_threshold: float = 0.7
    tier2_confidence_threshold: float = 0.6
    high_stakes_keywords: List[str] = field(default_factory=lambda: [
        "VA claim", "disability", "legal", "medical", "financial",
        "veteran", "benefits", "appeal", "nexus letter", "C&P exam",
        "service-connected", "TDIU", "rating decision",
    ])
    max_tier3_per_hour: int = 5
    enable_thermal_audit: bool = True

    @classmethod
    def from_dict(cls, d: Dict[str, Any]) -> "EscalationConfig":
        default_keywords = [
            "VA claim", "disability", "legal", "medical", "financial",
            "veteran", "benefits", "appeal", "nexus letter", "C&P exam",
            "service-connected", "TDIU", "rating decision",
        ]
        return cls(
            tier1_confidence_threshold=d.get("tier1_confidence_threshold", 0.7),
            tier2_confidence_threshold=d.get("tier2_confidence_threshold", 0.6),
            high_stakes_keywords=d.get("high_stakes_keywords", default_keywords),
            max_tier3_per_hour=d.get("max_tier3_per_hour", 5),
            enable_thermal_audit=d.get("enable_thermal_audit", True),
        )


@dataclass
class HarnessConfig:
    """Top-level harness configuration."""
    tier1: TierConfig = field(default_factory=TierConfig)
    tier2: TierConfig = field(default_factory=lambda: TierConfig(
        confidence_threshold=0.6,
        latency_target_ms=500,
        max_specialists=3,
    ))
    tier3: TierConfig = field(default_factory=lambda: TierConfig(
        confidence_threshold=0.0,
        latency_target_ms=120000,
        max_specialists=8,
    ))
    escalation: EscalationConfig = field(default_factory=EscalationConfig)

    @classmethod
    def from_dict(cls, d: Dict[str, Any]) -> "HarnessConfig":
        config = cls()
        if "tier1" in d:
            config.tier1 = TierConfig.from_dict(d["tier1"])
        if "tier2" in d:
            config.tier2 = TierConfig.from_dict(d["tier2"])
        if "tier3" in d:
            config.tier3 = TierConfig.from_dict(d["tier3"])
        if "escalation" in d:
            config.escalation = EscalationConfig.from_dict(d["escalation"])
        return config


def load_harness_config(config_path: Optional[str] = None) -> HarnessConfig:
    """Load harness configuration from YAML file.

    Resolution order:
    1. Explicit config_path argument
    2. HARNESS_CONFIG_PATH environment variable
    3. /ganuda/lib/harness/config.yaml (default)

    Returns default config if no file found (allows zero-config startup).
    """
    if config_path is None:
        config_path = os.environ.get(
            "HARNESS_CONFIG_PATH",
            "/ganuda/lib/harness/config.yaml"
        )

    if not os.path.exists(config_path):
        logger.warning(
            "Harness config not found at %s, using defaults", config_path
        )
        return HarnessConfig()

    try:
        with open(config_path, "r") as f:
            raw = yaml.safe_load(f) or {}

        # Resolve environment variable references in string values
        raw = _resolve_env_vars(raw)

        config = HarnessConfig.from_dict(raw)
        logger.info("Loaded harness config from %s", config_path)
        return config
    except Exception as e:
        logger.error("Failed to load harness config from %s: %s", config_path, e)
        return HarnessConfig()


def _resolve_env_vars(obj: Any) -> Any:
    """Recursively resolve ${ENV_VAR} references in config values."""
    if isinstance(obj, str) and obj.startswith("${") and obj.endswith("}"):
        env_name = obj[2:-1]
        return os.environ.get(env_name, obj)
    elif isinstance(obj, dict):
        return {k: _resolve_env_vars(v) for k, v in obj.items()}
    elif isinstance(obj, list):
        return [_resolve_env_vars(item) for item in obj]
    return obj
```

```python
"""
Escalation Engine -- Confidence and stakes-based tier promotion.

DC-6 Gradient Principle: the system finds its own level. Users never choose
tiers. Escalation is automatic, based on confidence scoring and stakes
detection. No de-escalation in v1.

Coyote rule: humans underestimate stakes. When in doubt, escalate.
Cost of over-escalation < cost of under-escalation.

Rate limiting: max N Tier 3 calls per hour per user (configurable).
Audit trail: every escalation decision is logged with full reasoning.
"""

import logging
import time
from collections import defaultdict
from typing import Optional, List, Dict, Any, Protocol

from lib.harness.core import (
    HarnessRequest,
    HarnessResponse,
    TierResult,
    StakesLevel,
)
from lib.harness.config import HarnessConfig, load_harness_config

logger = logging.getLogger("harness.escalation")


class TierHandler(Protocol):
    """Protocol for tier handlers.

    Each tier must implement this interface. The escalation engine calls
    handle() and uses the returned TierResult to decide whether to escalate.
    """

    def handle(
        self,
        request: HarnessRequest,
        prior_results: Optional[List[TierResult]] = None,
    ) -> TierResult:
        """Process a request and return a tier result."""
        ...


class EscalationEngine:
    """Routes requests through tiers based on confidence and stakes.

    The engine tries Tier 1 first. If confidence is below threshold,
    it escalates to Tier 2. If Tier 2 confidence is still low or
    stakes are detected as high, it escalates to Tier 3.

    Usage:
        engine = EscalationEngine(config)
        engine.register_tier(1, tier1_handler)
        engine.register_tier(2, tier2_handler)
        engine.register_tier(3, tier3_handler)
        response = engine.handle_request(request)
    """

    def __init__(self, config: Optional[HarnessConfig] = None):
        self.config = config or load_harness_config()
        self._tier_handlers: Dict[int, TierHandler] = {}
        # Rate limiting: {user_id: [timestamp, ...]}
        self._tier3_calls: Dict[str, List[float]] = defaultdict(list)

    def register_tier(self, tier_number: int, handler: TierHandler) -> None:
        """Register a handler for a specific tier (1, 2, or 3)."""
        if tier_number not in (1, 2, 3):
            raise ValueError(f"Invalid tier number: {tier_number}. Must be 1, 2, or 3.")
        self._tier_handlers[tier_number] = handler
        logger.info("Registered handler for Tier %d: %s", tier_number, type(handler).__name__)

    def handle_request(self, request: HarnessRequest) -> HarnessResponse:
        """Route a request through the tier chain with automatic escalation.

        This is the main entry point. It validates the request, then walks
        the tier chain until a tier returns sufficient confidence or we
        reach the highest registered tier.

        Args:
            request: The harness request to process.

        Returns:
            HarnessResponse with the final answer, tier used, and full
            escalation path.
        """
        start_time = time.time()

        # --- Input validation (Crawdad requirement) ---
        validation_errors = request.validate()
        if validation_errors:
            return HarnessResponse(
                answer="Request validation failed: " + "; ".join(validation_errors),
                tier_used=0,
                confidence=0.0,
                request_id=request.request_id,
                metadata={"errors": validation_errors},
            )

        # --- Force tier override (testing only) ---
        if request.force_tier is not None:
            logger.warning(
                "Force tier %d requested for %s (testing mode)",
                request.force_tier, request.request_id,
            )
            return self._execute_single_tier(
                request, request.force_tier, start_time
            )

        # --- Normal escalation flow ---
        results: List[TierResult] = []
        escalation_path: List[int] = []

        # TIER 1
        if 1 in self._tier_handlers:
            t1_result = self._execute_tier(request, 1, results)
            results.append(t1_result)
            escalation_path.append(1)

            if t1_result.confidence >= self.config.escalation.tier1_confidence_threshold:
                return self._build_response(
                    t1_result, escalation_path, start_time, request
                )
            logger.info(
                "Tier 1 confidence %.3f < threshold %.3f, escalating to Tier 2 [%s]",
                t1_result.confidence,
                self.config.escalation.tier1_confidence_threshold,
                request.request_id,
            )

        # TIER 2
        if 2 in self._tier_handlers:
            t2_result = self._execute_tier(request, 2, results)
            results.append(t2_result)
            escalation_path.append(2)

            stakes = self._detect_stakes(request, t2_result, results)
            t2_result.stakes = stakes

            if (
                t2_result.confidence >= self.config.escalation.tier2_confidence_threshold
                and stakes != StakesLevel.HIGH
            ):
                return self._build_response(
                    t2_result, escalation_path, start_time, request
                )
            escalation_reason = (
                "high_stakes" if stakes == StakesLevel.HIGH
                else "low_confidence"
            )
            logger.info(
                "Tier 2 escalating to Tier 3: reason=%s, confidence=%.3f, stakes=%s [%s]",
                escalation_reason, t2_result.confidence, stakes.value,
                request.request_id,
            )

        # TIER 3
        if 3 in self._tier_handlers:
            # Rate limit check
            if self._is_tier3_rate_limited(request.user_id):
                logger.warning(
                    "Tier 3 rate limited for user %s [%s]",
                    request.user_id, request.request_id,
                )
                # Return best result so far with rate limit notice
                best = results[-1] if results else TierResult(
                    tier=0, answer="Rate limited", confidence=0.0, latency_ms=0.0
                )
                resp = self._build_response(best, escalation_path, start_time, request)
                resp.metadata["rate_limited"] = True
                resp.metadata["rate_limit_message"] = (
                    "Tier 3 rate limited: max "
                    + str(self.config.escalation.max_tier3_per_hour) + "/hour"
                )
                return resp

            t3_result = self._execute_tier(request, 3, results)
            results.append(t3_result)
            escalation_path.append(3)
            self._record_tier3_call(request.user_id)
            return self._build_response(
                t3_result, escalation_path, start_time, request
            )

        # No tiers registered -- should not happen in production
        if not results:
            return HarnessResponse(
                answer="No tier handlers registered",
                tier_used=0,
                confidence=0.0,
                request_id=request.request_id,
                metadata={"error": "no_handlers"},
            )

        # Return whatever we have
        return self._build_response(
            results[-1], escalation_path, start_time, request
        )

    def _execute_tier(
        self,
        request: HarnessRequest,
        tier: int,
        prior_results: List[TierResult],
    ) -> TierResult:
        """Execute a single tier's handler with error handling."""
        handler = self._tier_handlers[tier]
        try:
            return handler.handle(request, prior_results if prior_results else None)
        except Exception as e:
            logger.error(
                "Tier %d handler failed for %s: %s",
                tier, request.request_id, e,
            )
            return TierResult(
                tier=tier,
                answer="Tier " + str(tier) + " processing error: " + str(e),
                confidence=0.0,
                latency_ms=0.0,
                metadata={"error": str(e)},
            )

    def _execute_single_tier(
        self,
        request: HarnessRequest,
        tier: int,
        start_time: float,
    ) -> HarnessResponse:
        """Execute a single forced tier (testing mode)."""
        if tier not in self._tier_handlers:
            return HarnessResponse(
                answer="Tier " + str(tier) + " handler not registered",
                tier_used=0,
                confidence=0.0,
                request_id=request.request_id,
                metadata={"error": "tier_" + str(tier) + "_not_registered"},
            )
        result = self._execute_tier(request, tier, [])
        return self._build_response(result, [tier], start_time, request)

    def _detect_stakes(
        self,
        request: HarnessRequest,
        current_result: TierResult,
        prior_results: List[TierResult],
    ) -> StakesLevel:
        """Detect stakes level from request content and tier results.

        Coyote rule: when in doubt, escalate.

        Checks:
        1. High-stakes keywords in query
        2. Confidence drop between tiers (getting less certain = risky)
        3. Low specialist agreement in Tier 2 (if applicable)
        """
        query_lower = request.query.lower()

        # Check high-stakes keywords
        for keyword in self.config.escalation.high_stakes_keywords:
            if keyword.lower() in query_lower:
                logger.info(
                    "High stakes detected: keyword '%s' found in query [%s]",
                    keyword, request.request_id,
                )
                return StakesLevel.HIGH

        # Check confidence drop between tiers
        if prior_results:
            last_confidence = prior_results[-1].confidence
            if (
                current_result.confidence < last_confidence
                and current_result.confidence < 0.5
            ):
                logger.info(
                    "High stakes detected: confidence dropped %.3f -> %.3f [%s]",
                    last_confidence, current_result.confidence, request.request_id,
                )
                return StakesLevel.HIGH

        # Check low diversity (specialist disagreement)
        if (
            current_result.diversity_score is not None
            and current_result.diversity_score > 0.8
        ):
            # High diversity = high disagreement = uncertain = medium stakes
            return StakesLevel.MEDIUM

        # Context-based stakes from request metadata
        if request.metadata.get("stakes_override") == "high":
            return StakesLevel.HIGH

        return StakesLevel.LOW

    def _build_response(
        self,
        final_result: TierResult,
        escalation_path: List[int],
        start_time: float,
        request: HarnessRequest,
    ) -> HarnessResponse:
        """Build the final HarnessResponse from tier results."""
        total_latency = (time.time() - start_time) * 1000  # ms
        return HarnessResponse(
            answer=final_result.answer,
            tier_used=final_result.tier,
            confidence=final_result.confidence,
            escalation_path=escalation_path,
            latency_ms=total_latency,
            diversity_score=final_result.diversity_score,
            council_vote_id=final_result.council_vote_id,
            request_id=request.request_id,
            metadata=final_result.metadata,
        )

    def _is_tier3_rate_limited(self, user_id: str) -> bool:
        """Check if user has exceeded Tier 3 rate limit."""
        now = time.time()
        hour_ago = now - 3600
        # Clean old entries
        self._tier3_calls[user_id] = [
            t for t in self._tier3_calls[user_id] if t > hour_ago
        ]
        return len(self._tier3_calls[user_id]) >= self.config.escalation.max_tier3_per_hour

    def _record_tier3_call(self, user_id: str) -> None:
        """Record a Tier 3 call for rate limiting."""
        self._tier3_calls[user_id].append(time.time())
```
