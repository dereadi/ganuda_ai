"""
Escalation Engine -- Confidence and stakes-based tier promotion.

DC-6 Gradient Principle: the system finds its own level. Users never choose
tiers. Escalation is automatic, based on confidence scoring and stakes
detection. No de-escalation in v1.

Coyote rule: humans underestimate stakes. When in doubt, escalate.
Cost of over-escalation < cost of under-escalation.

Rate limiting: max N Tier 3 calls per hour per user (configurable).
Audit trail: every escalation decision is logged with full reasoning.
"""

import logging
import time
from collections import defaultdict
from typing import Optional, List, Dict, Any, Protocol

from lib.harness.core import (
    HarnessRequest,
    HarnessResponse,
    TierResult,
    StakesLevel,
)
from lib.harness.config import HarnessConfig, load_harness_config

logger = logging.getLogger("harness.escalation")


class TierHandler(Protocol):
    """Protocol for tier handlers.

    Each tier must implement this interface. The escalation engine calls
    handle() and uses the returned TierResult to decide whether to escalate.
    """

    def handle(
        self,
        request: HarnessRequest,
        prior_results: Optional[List[TierResult]] = None,
    ) -> TierResult:
        """Process a request and return a tier result."""
        ...


class EscalationEngine:
    """Routes requests through tiers based on confidence and stakes.

    The engine tries Tier 1 first. If confidence is below threshold,
    it escalates to Tier 2. If Tier 2 confidence is still low or
    stakes are detected as high, it escalates to Tier 3.

    Usage:
        engine = EscalationEngine(config)
        engine.register_tier(1, tier1_handler)
        engine.register_tier(2, tier2_handler)
        engine.register_tier(3, tier3_handler)
        response = engine.handle_request(request)
    """

    def __init__(self, config: Optional[HarnessConfig] = None):
        self.config = config or load_harness_config()
        self._tier_handlers: Dict[int, TierHandler] = {}
        # Rate limiting: {user_id: [timestamp, ...]}
        self._tier3_calls: Dict[str, List[float]] = defaultdict(list)

    def register_tier(self, tier_number: int, handler: TierHandler) -> None:
        """Register a handler for a specific tier (1, 2, or 3)."""
        if tier_number not in (1, 2, 3):
            raise ValueError(f"Invalid tier number: {tier_number}. Must be 1, 2, or 3.")
        self._tier_handlers[tier_number] = handler
        logger.info("Registered handler for Tier %d: %s", tier_number, type(handler).__name__)

    def handle_request(self, request: HarnessRequest) -> HarnessResponse:
        """Route a request through the tier chain with automatic escalation.

        This is the main entry point. It validates the request, then walks
        the tier chain until a tier returns sufficient confidence or we
        reach the highest registered tier.

        Args:
            request: The harness request to process.

        Returns:
            HarnessResponse with the final answer, tier used, and full
            escalation path.
        """
        start_time = time.time()

        # --- Input validation (Crawdad requirement) ---
        validation_errors = request.validate()
        if validation_errors:
            return HarnessResponse(
                answer="Request validation failed: " + "; ".join(validation_errors),
                tier_used=0,
                confidence=0.0,
                request_id=request.request_id,
                metadata={"errors": validation_errors},
            )

        # --- Force tier override (testing only) ---
        if request.force_tier is not None:
            logger.warning(
                "Force tier %d requested for %s (testing mode)",
                request.force_tier, request.request_id,
            )
            return self._execute_single_tier(
                request, request.force_tier, start_time
            )

        # --- Normal escalation flow ---
        results: List[TierResult] = []
        escalation_path: List[int] = []

        # TIER 1
        if 1 in self._tier_handlers:
            t1_result = self._execute_tier(request, 1, results)
            results.append(t1_result)
            escalation_path.append(1)

            if t1_result.confidence >= self.config.escalation.tier1_confidence_threshold:
                return self._build_response(
                    t1_result, escalation_path, start_time, request
                )
            logger.info(
                "Tier 1 confidence %.3f < threshold %.3f, escalating to Tier 2 [%s]",
                t1_result.confidence,
                self.config.escalation.tier1_confidence_threshold,
                request.request_id,
            )

        # TIER 2
        if 2 in self._tier_handlers:
            t2_result = self._execute_tier(request, 2, results)
            results.append(t2_result)
            escalation_path.append(2)

            stakes = self._detect_stakes(request, t2_result, results)
            t2_result.stakes = stakes

            if (
                t2_result.confidence >= self.config.escalation.tier2_confidence_threshold
                and stakes != StakesLevel.HIGH
            ):
                return self._build_response(
                    t2_result, escalation_path, start_time, request
                )
            escalation_reason = (
                "high_stakes" if stakes == StakesLevel.HIGH
                else "low_confidence"
            )
            logger.info(
                "Tier 2 escalating to Tier 3: reason=%s, confidence=%.3f, stakes=%s [%s]",
                escalation_reason, t2_result.confidence, stakes.value,
                request.request_id,
            )

        # TIER 3
        if 3 in self._tier_handlers:
            # Rate limit check
            if self._is_tier3_rate_limited(request.user_id):
                logger.warning(
                    "Tier 3 rate limited for user %s [%s]",
                    request.user_id, request.request_id,
                )
                # Return best result so far with rate limit notice
                best = results[-1] if results else TierResult(
                    tier=0, answer="Rate limited", confidence=0.0, latency_ms=0.0
                )
                resp = self._build_response(best, escalation_path, start_time, request)
                resp.metadata["rate_limited"] = True
                resp.metadata["rate_limit_message"] = (
                    "Tier 3 rate limited: max "
                    + str(self.config.escalation.max_tier3_per_hour) + "/hour"
                )
                return resp

            t3_result = self._execute_tier(request, 3, results)
            results.append(t3_result)
            escalation_path.append(3)
            self._record_tier3_call(request.user_id)
            return self._build_response(
                t3_result, escalation_path, start_time, request
            )

        # No tiers registered -- should not happen in production
        if not results:
            return HarnessResponse(
                answer="No tier handlers registered",
                tier_used=0,
                confidence=0.0,
                request_id=request.request_id,
                metadata={"error": "no_handlers"},
            )

        # Return whatever we have
        return self._build_response(
            results[-1], escalation_path, start_time, request
        )

    def _execute_tier(
        self,
        request: HarnessRequest,
        tier: int,
        prior_results: List[TierResult],
    ) -> TierResult:
        """Execute a single tier's handler with error handling."""
        handler = self._tier_handlers[tier]
        try:
            return handler.handle(request, prior_results if prior_results else None)
        except Exception as e:
            logger.error(
                "Tier %d handler failed for %s: %s",
                tier, request.request_id, e,
            )
            return TierResult(
                tier=tier,
                answer="Tier " + str(tier) + " processing error: " + str(e),
                confidence=0.0,
                latency_ms=0.0,
                metadata={"error": str(e)},
            )

    def _execute_single_tier(
        self,
        request: HarnessRequest,
        tier: int,
        start_time: float,
    ) -> HarnessResponse:
        """Execute a single forced tier (testing mode)."""
        if tier not in self._tier_handlers:
            return HarnessResponse(
                answer="Tier " + str(tier) + " handler not registered",
                tier_used=0,
                confidence=0.0,
                request_id=request.request_id,
                metadata={"error": "tier_" + str(tier) + "_not_registered"},
            )
        result = self._execute_tier(request, tier, [])
        return self._build_response(result, [tier], start_time, request)

    def _detect_stakes(
        self,
        request: HarnessRequest,
        current_result: TierResult,
        prior_results: List[TierResult],
    ) -> StakesLevel:
        """Detect stakes level from request content and tier results.

        Coyote rule: when in doubt, escalate.

        Checks:
        1. High-stakes keywords in query
        2. Confidence drop between tiers (getting less certain = risky)
        3. Low specialist agreement in Tier 2 (if applicable)
        """
        query_lower = request.query.lower()

        # Check high-stakes keywords
        for keyword in self.config.escalation.high_stakes_keywords:
            if keyword.lower() in query_lower:
                logger.info(
                    "High stakes detected: keyword '%s' found in query [%s]",
                    keyword, request.request_id,
                )
                return StakesLevel.HIGH

        # Check confidence drop between tiers
        if prior_results:
            last_confidence = prior_results[-1].confidence
            if (
                current_result.confidence < last_confidence
                and current_result.confidence < 0.5
            ):
                logger.info(
                    "High stakes detected: confidence dropped %.3f -> %.3f [%s]",
                    last_confidence, current_result.confidence, request.request_id,
                )
                return StakesLevel.HIGH

        # Check low diversity (specialist disagreement)
        if (
            current_result.diversity_score is not None
            and current_result.diversity_score > 0.8
        ):
            # High diversity = high disagreement = uncertain = medium stakes
            return StakesLevel.MEDIUM

        # Context-based stakes from request metadata
        if request.metadata.get("stakes_override") == "high":
            return StakesLevel.HIGH

        return StakesLevel.LOW

    def _build_response(
        self,
        final_result: TierResult,
        escalation_path: List[int],
        start_time: float,
        request: HarnessRequest,
    ) -> HarnessResponse:
        """Build the final HarnessResponse from tier results."""
        total_latency = (time.time() - start_time) * 1000  # ms
        return HarnessResponse(
            answer=final_result.answer,
            tier_used=final_result.tier,
            confidence=final_result.confidence,
            escalation_path=escalation_path,
            latency_ms=total_latency,
            diversity_score=final_result.diversity_score,
            council_vote_id=final_result.council_vote_id,
            request_id=request.request_id,
            metadata=final_result.metadata,
        )

    def _is_tier3_rate_limited(self, user_id: str) -> bool:
        """Check if user has exceeded Tier 3 rate limit."""
        now = time.time()
        hour_ago = now - 3600
        # Clean old entries
        self._tier3_calls[user_id] = [
            t for t in self._tier3_calls[user_id] if t > hour_ago
        ]
        return len(self._tier3_calls[user_id]) >= self.config.escalation.max_tier3_per_hour

    def _record_tier3_call(self, user_id: str) -> None:
        """Record a Tier 3 call for rate limiting."""
        self._tier3_calls[user_id].append(time.time())
